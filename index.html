    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        import { getFirestore, collection, addDoc, onSnapshot, query, doc, setDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        // Firebase Environment Setup

        const firebaseConfig = JSON.parse(__firebase_config);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'thesale3-camping';

        const app = initializeApp(firebaseConfig);

        const auth = getAuth(app);

        const db = getFirestore(app);


        let products = [];

        let user = null;


        // AUTHENTICATION (RULE 3)

        const initAuth = async () => {

            try {

                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                    await signInWithCustomToken(auth, __initial_auth_token);

                } else {

                    await signInAnonymously(auth);

                }

            } catch (error) {

                console.error("Auth failed:", error);

            }

        };


        // Listen for auth state

        onAuthStateChanged(auth, (authenticatedUser) => {

            user = authenticatedUser;

            if (user) {

                loadProducts();

            }

        });


        // LOAD PRODUCTS (RULE 1 & 2)

        const loadProducts = () => {

            if (!user) return;

            // Strict path requirement for artifacts

            const productCollection = collection(db, 'artifacts', appId, 'public', 'data', 'products');

            

            // Simple query only (no complex filters/orders)

            onSnapshot(productCollection, (snapshot) => {

                products = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

                

                // Seed initial data if empty

                if (products.length === 0) {

                    seedInitialProducts();

                } else {

                    renderProducts(products);

                }

            }, (error) => {

                console.error("Firestore Error:", error);

                document.getElementById('loadingMessage').innerText = "Permission Error: Check console for details.";

            });

        };


        const seedInitialProducts = async () => {

            if (!user) return;

            const productCollection = collection(db, 'artifacts', appId, 'public', 'data', 'products');

            const initial = [

                { name: "ເຄື່ອງບົດກາເຟ Coffee grinder", category: "coffee", price: 195000, description: "ເຄື່ອງບົດກາເຟແບບມືໝຸນ ຄຸນນະພາບສູງ ໃຊ້ງານງ່າຍ.", image: "https://i.postimg.cc/vZNN0NDm/138558955-160501135864025-1071256138491929297-n.jpg" },

                { name: "Nomad Compact Stove", category: "cooking", price: 54500, description: "Pocket-sized gas stove with precision flame control.", image: "https://images.unsplash.com/photo-1523348837708-15d4a09cfac2?auto=format&fit=crop&q=80&w=800" }

            ];

            

            for (const p of initial) {

                try {

                    await addDoc(productCollection, p);

                } catch (e) {

                    console.error("Seeding error:", e);

                }

            }

        };


        // UI FUNCTIONS

        window.renderProducts = (items) => {

            const grid = document.getElementById('productGrid');

            if (!grid) return;


            if (items.length === 0) {

                grid.innerHTML = `<div class="col-span-full py-20 text-gray-400 uppercase tracking-widest text-xs font-bold">No products found.</div>`;

                return;

            }


            grid.innerHTML = items.map(p => `

                <div class="product-card group overflow-hidden flex flex-col cursor-pointer fade-in" onclick="openModal('${p.id}')">

                    <div class="h-80 overflow-hidden relative">

                        <img src="${p.image}" class="w-full h-full object-cover group-hover:scale-105 transition-transform duration-1000" onerror="this.src='https://via.placeholder.com/400x400?text=Image+Missing'">

                    </div>

                    <div class="p-10 flex flex-col flex-grow">

                        <span class="text-[10px] font-bold text-gray-400 uppercase tracking-[0.2em] mb-3">${p.category}</span>

                        <h3 class="font-bold text-2xl mb-2 text-gray-900 group-hover:text-emerald-700 transition-colors tracking-tighter leading-tight">${p.name}</h3>

                        <p class="text-xl font-bold text-emerald-600 mb-10 italic">LAK ${Number(p.price).toLocaleString()}</p>

                        <div class="mt-auto inline-flex items-center justify-center gap-3 text-[11px] font-bold uppercase tracking-widest text-gray-900 group-hover:gap-5 transition-all">

                            View Gear <i class="fa-solid fa-arrow-right-long text-emerald-600"></i>

                        </div>

                    </div>

                </div>

            `).join('');

        };


        window.filterProducts = (cat, el) => {

            document.querySelectorAll('.category-pill').forEach(b => {

                b.classList.remove('active', 'bg-emerald-600', 'text-white');

                b.classList.add('bg-white', 'text-gray-400', 'border-gray-100');

            });

            el.classList.add('active', 'bg-emerald-600', 'text-white');

            el.classList.remove('bg-white', 'text-gray-400', 'border-gray-100');

            

            // Client-side filtering (RULE 2)

            const filtered = cat === 'all' ? products : products.filter(p => p.category === cat);

            renderProducts(filtered);

        };


        window.searchProducts = () => {

            const queryText = document.getElementById('searchInput').value.toLowerCase();

            const filtered = products.filter(p => p.name.toLowerCase().includes(queryText));

            renderProducts(filtered);

        };


        window.openModal = (id) => {

            const p = products.find(x => x.id === id);

            if (!p) return;

            document.getElementById('modalTitle').innerText = p.name;

            document.getElementById('modalCategory').innerText = p.category;

            document.getElementById('modalPrice').innerText = `LAK ${Number(p.price).toLocaleString()}`;

            document.getElementById('modalDescription').innerText = p.description;

            document.getElementById('modalImage').src = p.image;

            document.getElementById('whatsappBtn').href = `https://wa.me/8562054849393?text=Sabaidee! I am interested in: ${p.name}`;

            document.getElementById('modal').style.display = 'flex';

            document.body.style.overflow = 'hidden';

        };


        window.closeModal = () => {

            document.getElementById('modal').style.display = 'none';

            document.getElementById('adminModal').style.display = 'none';

            document.body.style.overflow = 'auto';

        };


        window.toggleAdmin = () => {

            const modal = document.getElementById('adminModal');

            modal.style.display = modal.style.display === 'flex' ? 'none' : 'flex';

        };


        // ADMIN FORM HANDLING

        document.getElementById('adminForm').onsubmit = async (e) => {

            e.preventDefault();

            if (!user) {

                console.error("Not authenticated");

                return;

            }

            

            const newProduct = {

                name: document.getElementById('pName').value,

                price: Number(document.getElementById('pPrice').value),

                category: document.getElementById('pCategory').value,

                image: document.getElementById('pImage').value,

                description: document.getElementById('pDesc').value,

                createdAt: new Date().toISOString()

            };


            try {

                const productCollection = collection(db, 'artifacts', appId, 'public', 'data', 'products');

                await addDoc(productCollection, newProduct);

                document.getElementById('adminForm').reset();

                toggleAdmin();

            } catch (err) {

                console.error("Save Error:", err);

            }

        };


        // Start Auth process

        initAuth();
